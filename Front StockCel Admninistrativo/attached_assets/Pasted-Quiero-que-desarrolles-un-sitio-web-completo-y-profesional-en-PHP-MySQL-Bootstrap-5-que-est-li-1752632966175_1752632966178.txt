🗄️ BASE DE DATOS (estructura.sql)
======================================

CREATE DATABASE softwarepar;
USE softwarepar;

CREATE TABLE usuarios (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nombre VARCHAR(100),
  email VARCHAR(100) UNIQUE,
  telefono VARCHAR(20),
  empresa VARCHAR(100),
  password VARCHAR(255),
  rol ENUM('admin', 'revendedor') DEFAULT 'revendedor',
  fecha_registro DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE licencias (
  id INT AUTO_INCREMENT PRIMARY KEY,
  id_usuario INT,
  producto VARCHAR(100),
  codigo_licencia VARCHAR(100) UNIQUE,
  estado ENUM('activa', 'vencida', 'suspendida') DEFAULT 'activa',
  fecha_emision DATETIME DEFAULT CURRENT_TIMESTAMP,
  fecha_vencimiento DATE,
  observaciones TEXT,
  FOREIGN KEY (id_usuario) REFERENCES usuarios(id)
);

CREATE TABLE clientes_finales (
  id INT AUTO_INCREMENT PRIMARY KEY,
  id_licencia INT,
  nombre_cliente VARCHAR(100),
  email_cliente VARCHAR(100),
  telefono_cliente VARCHAR(20),
  domicilio VARCHAR(150),
  fecha_alta DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (id_licencia) REFERENCES licencias(id)
);
